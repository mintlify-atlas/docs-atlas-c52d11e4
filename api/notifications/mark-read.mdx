<Warning>
  The `notification.mark_read` method is **not yet implemented** in the v2 socket API. This documentation describes the planned interface and behavior.
</Warning>

## Status

Currently, notifications can only be marked as read through:

1. **Automatic marking** - When user focuses the associated surface
2. **UI interactions** - Clicking notifications in the panel
3. **Keyboard shortcuts** - Using notification panel shortcuts

A programmatic API method is planned for future release.

### Existing Internal Methods

The internal Swift API has mark_read methods:

From `Sources/TerminalNotificationStore.swift:222-262`:

```swift
// Mark single notification by ID
func markRead(id: UUID) {
    var updated = notifications
    guard let index = updated.firstIndex(where: { $0.id == id }) else { return }
    guard !updated[index].isRead else { return }
    updated[index].isRead = true
    notifications = updated
    center.removeDeliveredNotifications(withIdentifiers: [id.uuidString])
}

// Mark all notifications for workspace
func markRead(forTabId tabId: UUID) {
    var updated = notifications
    var idsToClear: [String] = []
    for index in updated.indices {
        if updated[index].tabId == tabId && !updated[index].isRead {
            updated[index].isRead = true
            idsToClear.append(updated[index].id.uuidString)
        }
    }
    if !idsToClear.isEmpty {
        notifications = updated
        center.removeDeliveredNotifications(withIdentifiers: idsToClear)
    }
}

// Mark all for specific surface
func markRead(forTabId tabId: UUID, surfaceId: UUID?) {
    var updated = notifications
    var idsToClear: [String] = []
    for index in updated.indices {
        if updated[index].tabId == tabId,
           updated[index].surfaceId == surfaceId,
           !updated[index].isRead {
            updated[index].isRead = true
            idsToClear.append(updated[index].id.uuidString)
        }
    }
    if !idsToClear.isEmpty {
        notifications = updated
        center.removeDeliveredNotifications(withIdentifiers: idsToClear)
        center.removePendingNotificationRequests(withIdentifiers: idsToClear)
    }
}
```

These methods are used internally but not yet exposed via the v2 socket API.

### Automatic Read Behavior

Notifications are automatically marked as read when:

1. **User focuses surface** - Switching to the panel with `Cmd+Number` or clicking
2. **Workspace becomes active** - Selecting workspace in sidebar
3. **Panel interaction** - Opening notification panel and viewing notification

From `Sources/Workspace.swift:1460`:

```swift
AppDelegate.shared?.notificationStore?.markRead(forTabId: id, surfaceId: panelId)
```

From `Sources/TabManager.swift:1575`:

```swift
notificationStore.markRead(forTabId: tabId, surfaceId: panelId)
```

### Read Status Impact

- **Dock badge** - Unread count decreases
- **Workspace badges** - Per-workspace unread count updates
- **Panel indicators** - Blue dot removed from notification
- **System notifications** - Delivered notification removed from macOS notification center

## Future Implementation

The planned implementation would add this v2 method handler:

```swift
// In Sources/TerminalController.swift
case "notification.mark_read":
    return v2Result(id: id, self.v2NotificationMarkRead(params: params))

private func v2NotificationMarkRead(params: [String: Any]) -> V2CallResult {
    if let notificationId = v2UUID(params, "notification_id") {
        DispatchQueue.main.sync {
            TerminalNotificationStore.shared.markRead(id: notificationId)
        }
        return .ok(["marked": 1])
    }
    
    if let all = params["all"] as? Bool, all {
        var count = 0
        DispatchQueue.main.sync {
            count = TerminalNotificationStore.shared.notifications.filter { !$0.isRead }.count
            TerminalNotificationStore.shared.markAllRead()
        }
        return .ok(["marked": count])
    }
    
    if let workspaceId = v2UUID(params, "workspace_id") {
        if let surfaceId = v2UUID(params, "surface_id") {
            DispatchQueue.main.sync {
                TerminalNotificationStore.shared.markRead(forTabId: workspaceId, surfaceId: surfaceId)
            }
        } else {
            DispatchQueue.main.sync {
                TerminalNotificationStore.shared.markRead(forTabId: workspaceId)
            }
        }
        return .ok([:]) 
    }
    
    return .err(code: "invalid_params", message: "Missing required parameters", data: nil)
}
```

## Use Cases for Future API

### Mark After Processing

```javascript
// Get unread notifications
const listResult = await sendSocketRequest({
  method: "notification.list",
  params: {}
});

const unread = listResult.result.notifications.filter(n => !n.is_read);

// Process each
for (const notif of unread) {
  await processNotification(notif);
  
  // Mark as read after processing
  await sendSocketRequest({
    method: "notification.mark_read",
    params: {
      notification_id: notif.id
    }
  });
}
```

### Bulk Mark by Workspace

```javascript
// Mark all notifications for workspace as read
await sendSocketRequest({
  method: "notification.mark_read",
  params: {
    workspace_id: "a1b2c3d4-e5f6-4a7b-8c9d-0e1f2a3b4c5d"
  }
});
```

### Mark All Read

```javascript
// Mark all notifications as read
await sendSocketRequest({
  method: "notification.mark_read",
  params: {
    all: true
  }
});
```
